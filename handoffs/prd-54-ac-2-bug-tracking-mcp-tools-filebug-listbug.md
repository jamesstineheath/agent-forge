## Context

Agent Forge has an MCP (Model Context Protocol) server at `lib/mcp/server.ts` that exposes tools to AI assistants. Work item tools already exist at `lib/mcp/tools/work-items.ts` and serve as the pattern to follow.

This task adds three bug-tracking MCP tools backed by a Notion Bugs database (data source ID: `023f3621-2885-468d-a8cf-2e0bd1458bb3`). The Notion client is available via `getClient()` from `lib/notion.ts`.

The Bugs database schema fields:
- **Title** (title property)
- **Status** (select): Open, Triaged, In Progress, Fixed, Won't Fix, Reopened
- **Severity** (select): Critical, High, Medium, Low
- **Target Repo** (select)
- **Source** (select): Manual, Agent, etc.
- **Related PRD** (relation)
- **Work Item ID** (rich_text)
- **Fix PR URL** (url)
- **Context** (rich_text)
- **Affected Files** (rich_text)
- **Bug ID** (unique_id — auto-generated by Notion, read-only, format: BUG-N)

The Notion API is already integrated. Look at how `lib/notion.ts` uses `getClient()` and how `lib/mcp/tools/work-items.ts` structures tool registration to match those patterns exactly.

## Requirements

1. Create `lib/mcp/tools/bugs.ts` exporting `registerBugTools(server)` following the pattern in `lib/mcp/tools/work-items.ts`
2. Implement `file_bug` tool: accepts title, severity (Critical/High/Medium/Low), target_repo, context, and optional affected_files; creates a page in the Bugs database with Status="Triaged" and Source="Manual"; returns created page ID and Bug ID (BUG-N)
3. Implement `list_bugs` tool: accepts optional severity and target_repo filters; queries the Bugs database excluding Fixed and Won't Fix statuses; returns title, status, severity, repo, bug ID, and created time
4. Implement `update_bug_status` tool: accepts page_id and status (Open/Triaged/In Progress/Fixed/Won't Fix/Reopened); optionally accepts fix_pr_url; updates the Status select and optionally Fix PR URL on the Notion page
5. Register all three tools in `lib/mcp/server.ts` by importing and calling `registerBugTools(server)`
6. TypeScript must compile with no errors (`npx tsc --noEmit`)
7. The Bugs database ID must be resolved from data source ID `023f3621-2885-468d-a8cf-2e0bd1458bb3` — check `lib/notion.ts` for how data sources are looked up, or hardcode the database ID if a lookup utility doesn't exist

Create the file following the exact structure of `lib/mcp/tools/work-items.ts`. Below is the full implementation to adapt:

```typescript
import { getClient } from "../../notion";
// Import any data source lookup utility if it exists, e.g.:
// import { getDataSourceDatabaseId } from "../../notion";

const BUGS_DB_DATA_SOURCE_ID = "023f3621-2885-468d-a8cf-2e0bd1458bb3";

// If notion.ts has a lookup function, use it inside handlers.
// Otherwise resolve the database ID once at module level via a helper.

type SeverityOption = "Critical" | "High" | "Medium" | "Low";
type StatusOption = "Open" | "Triaged" | "In Progress" | "Fixed" | "Won't Fix" | "Reopened";

export function registerBugTools(server: any) {
  // ── file_bug ──────────────────────────────────────────────────────────────
  server.tool(
    "file_bug",
    "Create a new bug report in the Notion Bugs database. Returns the created page ID and BUG-N unique ID.",
    {
      type: "object",
      properties: {
        title: {
          type: "string",
          description: "Short descriptive title for the bug",
        },
        severity: {
          type: "string",
          enum: ["Critical", "High", "Medium", "Low"],
          description: "Bug severity level",
        },
        target_repo: {
          type: "string",
          description: "The repository where this bug was observed (e.g. jamesstineheath/agent-forge)",
        },
        context: {
          type: "string",
          description: "Detailed description of the bug, steps to reproduce, observed vs expected behavior",
        },
        affected_files: {
          type: "string",
          description: "Optional comma-separated list of affected file paths",
        },
      },
      required: ["title", "severity", "target_repo", "context"],
    },
    async (args: {
      title: string;
      severity: SeverityOption;
      target_repo: string;
      context: string;
      affected_files?: string;
    }) => {
      try {
        const notion = getClient();
        // Resolve database ID
        const databaseId = await resolveBugsDatabaseId();

        const properties: Record<string, any> = {
          Title: {
            title: [{ text: { content: args.title } }],
          },
          Status: {
            select: { name: "Triaged" },
          },
          Severity: {
            select: { name: args.severity },
          },
          "Target Repo": {
            select: { name: args.target_repo },
          },
          Source: {
            select: { name: "Manual" },
          },
          Context: {
            rich_text: [{ text: { content: args.context } }],
          },
        };

        if (args.affected_files) {
          properties["Affected Files"] = {
            rich_text: [{ text: { content: args.affected_files } }],
          };
        }

        const page = await notion.pages.create({
          parent: { database_id: databaseId },
          properties,
        });

        // Extract the Bug ID (unique_id) from the created page
        const bugIdProp = (page as any).properties?.["Bug ID"];
        const bugId = bugIdProp?.unique_id
          ? `BUG-${bugIdProp.unique_id.number}`
          : "BUG-?";

        return {
          content: [
            {
              type: "text",
              text: JSON.stringify({
                success: true,
                page_id: page.id,
                bug_id: bugId,
                message: `Bug filed successfully as ${bugId}`,
              }),
            },
          ],
        };
      } catch (error: any) {
        return {
          content: [
            {
              type: "text",
              text: JSON.stringify({
                success: false,
                error: error?.message ?? String(error),
              }),
            },
          ],
          isError: true,
        };
      }
    }
  );

  // ── list_bugs ─────────────────────────────────────────────────────────────
  server.tool(
    "list_bugs",
    "List bugs from the Notion Bugs database. Excludes Fixed and Won't Fix by default. Optionally filter by severity and/or target repo.",
    {
      type: "object",
      properties: {
        severity: {
          type: "string",
          enum: ["Critical", "High", "Medium", "Low"],
          description: "Optional: filter by severity",
        },
        target_repo: {
          type: "string",
          description: "Optional: filter by target repository",
        },
      },
      required: [],
    },
    async (args: { severity?: SeverityOption; target_repo?: string }) => {
      try {
        const notion = getClient();
        const databaseId = await resolveBugsDatabaseId();

        // Build filter: exclude Fixed and Won't Fix, optionally add severity/repo filters
        const andFilters: any[] = [
          {
            property: "Status",
            select: { does_not_equal: "Fixed" },
          },
          {
            property: "Status",
            select: { does_not_equal: "Won't Fix" },
          },
        ];

        if (args.severity) {
          andFilters.push({
            property: "Severity",
            select: { equals: args.severity },
          });
        }

        if (args.target_repo) {
          andFilters.push({
            property: "Target Repo",
            select: { equals: args.target_repo },
          });
        }

        const response = await notion.databases.query({
          database_id: databaseId,
          filter: andFilters.length === 1 ? andFilters[0] : { and: andFilters },
          sorts: [{ timestamp: "created_time", direction: "descending" }],
          page_size: 50,
        });

        const bugs = response.results.map((page: any) => {
          const props = page.properties ?? {};
          const titleArr = props.Title?.title ?? [];
          const title = titleArr.map((t: any) => t.plain_text).join("");
          const status = props.Status?.select?.name ?? null;
          const severity = props.Severity?.select?.name ?? null;
          const repo = props["Target Repo"]?.select?.name ?? null;
          const bugIdProp = props["Bug ID"]?.unique_id;
          const bugId = bugIdProp ? `BUG-${bugIdProp.number}` : null;

          return {
            page_id: page.id,
            bug_id: bugId,
            title,
            status,
            severity,
            target_repo: repo,
            created_time: page.created_time,
          };
        });

        return {
          content: [
            {
              type: "text",
              text: JSON.stringify({ success: true, count: bugs.length, bugs }),
            },
          ],
        };
      } catch (error: any) {
        return {
          content: [
            {
              type: "text",
              text: JSON.stringify({
                success: false,
                error: error?.message ?? String(error),
              }),
            },
          ],
          isError: true,
        };
      }
    }
  );

  // ── update_bug_status ─────────────────────────────────────────────────────
  server.tool(
    "update_bug_status",
    "Update the status of a bug in the Notion Bugs database. Optionally set the Fix PR URL when marking as Fixed.",
    {
      type: "object",
      properties: {
        page_id: {
          type: "string",
          description: "The Notion page ID of the bug to update",
        },
        status: {
          type: "string",
          enum: ["Open", "Triaged", "In Progress", "Fixed", "Won't Fix", "Reopened"],
          description: "New status for the bug",
        },
        fix_pr_url: {
          type: "string",
          description: "Optional: URL of the PR that fixes this bug (populates Fix PR URL field)",
        },
      },
      required: ["page_id", "status"],
    },
    async (args: {
      page_id: string;
      status: StatusOption;
      fix_pr_url?: string;
    }) => {
      try {
        const notion = getClient();

        const properties: Record<string, any> = {
          Status: {
            select: { name: args.status },
          },
        };

        if (args.fix_pr_url) {
          properties["Fix PR URL"] = {
            url: args.fix_pr_url,
          };
        }

        await notion.pages.update({
          page_id: args.page_id,
          properties,
        });

        return {
          content: [
            {
              type: "text",
              text: JSON.stringify({
                success: true,
                page_id: args.page_id,
                status: args.status,
                message: `Bug status updated to "${args.status}"`,
              }),
            },
          ],
        };
      } catch (error: any) {
        return {
          content: [
            {
              type: "text",
              text: JSON.stringify({
                success: false,
                error: error?.message ?? String(error),
              }),
            },
          ],
          isError: true,
        };
      }
    }
  );
}
```
